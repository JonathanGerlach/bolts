    Module bolts.traits

    Provides utilites that allow you to determine type traits

    Example

    class C {}
struct S {}
struct S1 {
    this(typeof(null)) {}
    void opAssign(typeof(null)) {}
}

static assert( isNullable!C);
static assert(!isNullable!S);
static assert( isNullable!S1);
static assert( isNullable!(int *));
static assert(!isNullable!(int));
